> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Build a realistic Folketing election night demo scenario (`fv2022`) using real FV2022 vote results and FV2026 geography/candidates, with 33 waves that simulate election night progression from small islands to large cities.

**Architecture:** A one-time build script (`scripts/build_fv2022_scenario.py`) downloads FV2026 geography + candidates from SFTP and FV2022 vote results from the valg.dk API, joins them by normalised name, and writes pre-baked wave directories to `valg/scenarios/fv2022/`. A scenario module (`valg/scenarios/fv2022.py`) reads those waves at runtime. Two fixes to `valg/demo.py` are required first: fix kandidat-data processing (currently silently dropped), and fix restart cleanup to remove scenario-specific output dirs.

**Tech Stack:** Python 3.11+, paramiko (SFTP), urllib (HTTP), json, csv, pathlib, pytest. No new dependencies.

```python
"""
FV2022 demo scenario: replay Folketing election 2022 results as election night.

Pre-baked wave bundles live in valg/scenarios/fv2022/wave_NN/.
Each wave's files are copied into valg-data/demo/fv2022/ at runtime.

Geography and candidates are from FV2026 (latest available).
Vote results are real FV2022 data from the valg.dk public API.
"""
from __future__ import annotations

import json
import shutil
from pathlib import Path

from valg.demo import Scenario, Step

_WAVE_DIR = Path(__file__).parent / "fv2022"
_DEST_SUBPATH = Path("demo") / "fv2022"


def _copy_wave(wave_dir: Path, data_repo: Path) -> list[Path]:
    """Copy all non-meta files from wave_dir into data_repo/demo/fv2022/."""
    dest_base = data_repo / _DEST_SUBPATH
    written: list[Path] = []
    for src in wave_dir.rglob("*"):
        if src.is_dir() or src.name == "_meta.json":
            continue
        relative = src.relative_to(wave_dir)
        dest = dest_base / relative
        dest.parent.mkdir(parents=True, exist_ok=True)
        shutil.copy2(src, dest)
        written.append(dest)
    return written


def make_steps(wave_dir: Path, data_repo: Path) -> list[Step]:
    """Build Step list from pre-baked wave directories."""
    steps = []
    for wd in sorted(wave_dir.glob("wave_*")):
        meta_path = wd / "_meta.json"
        if not meta_path.exists():
            continue
        meta = json.loads(meta_path.read_text())
        steps.append(Step(
            name=meta["label"],
            wave=None,
            setup=meta.get("phase") == "setup",
            process=True,
            commit=True,
            base_interval_s=float(meta["interval_s"]),
            write_fn=lambda d, src=wd: _copy_wave(src, d),
        ))
    return steps


FV2022_SCENARIO = Scenario(
    name="FV2022 — Folketing 1. november 2022",
    description="Rigtige stemmeresultater fra Folketingsvalget 2022, afspillet som valgaften.",
    steps=[],
    steps_factory=lambda data_repo: make_steps(_WAVE_DIR, data_repo),
    output_dir=_DEST_SUBPATH,
)
```

```python
def normalize_name(s: str) -> str:
    """Lowercase, strip diacritics, collapse whitespace.

    Handles Danish characters: Æ/æ→ae, Ø/ø→o, Å/å→a explicitly before NFD
    decomposition, since these are not decomposable via NFD alone.
    """
    s = s.strip().lower()
    # Explicit Danish romanisation before NFD (Æ/Ø/Å not decomposable via NFD)
    s = s.replace("æ", "ae").replace("ø", "oe").replace("å", "aa")
    # Decompose remaining unicode, drop combining characters (strips ´ ¨ etc.)
    s = "".join(
        c for c in unicodedata.normalize("NFD", s)
        if unicodedata.category(c) != "Mn"
    )
    # Collapse internal whitespace
    s = " ".join(s.split())
    return s


def distribute_candidate_votes(party_votes: int, candidates: list[dict]) -> list[int]:
    """
    Distribute party_votes to candidates using 35% kredskandidat + power decay.

    Returns list of integer vote counts (same length as candidates, same order).
    """
    if not candidates or party_votes == 0:
        return [0] * len(candidates)

    n = len(candidates)
    if n == 1:
        return [party_votes]

    # Position 1 gets 35% floor
    pos1_votes = int(party_votes * 0.35)
    remainder = party_votes - pos1_votes

    # Remaining candidates (positions 2..n) get weight 1/pos^0.7
    weights = [1.0 / (i ** 0.7) for i in range(2, n + 1)]
    total_w = sum(weights)

    dist = [pos1_votes]
    allocated = pos1_votes
    for i, w in enumerate(weights):
        v = int(remainder * w / total_w)
        dist.append(v)
        allocated += v

    # Assign rounding remainder to position 1
    dist[0] += party_votes - allocated
    return dist


def build_partistemmefordeling(ok_id: str, party_totals: dict[str, int]) -> dict:
    """Build a partistemmefordeling JSON dict for one opstillingskreds."""
    return {
        "Valg": {
            "OpstillingskredsId": str(ok_id),
            "Partier": [
                {"PartiId": party_id, "Stemmer": votes}
                for party_id, votes in sorted(party_totals.items())
            ],
        }
    }


def build_valgresultater(
    ao_id: str,
    optaellingstype: str,
    party_data: dict[str, dict],
    ao_ok_id: str,
) -> dict:
    """
    Build a valgresultater JSON dict for one afstemningsomraade.

    party_data: {party_id: {"total": int, "candidates_by_ok": {ok_id: [candidate dicts]}}}
    ao_ok_id: the opstillingskreds this AO belongs to (for candidate lookup)
    """
    inden_for_parti = []
    for party_id, pdata in sorted(party_data.items()):
        total = pdata.get("total", 0)
        candidates = pdata.get("candidates_by_ok", {}).get(str(ao_ok_id), [])
        candidate_votes = distribute_candidate_votes(total, candidates)
        kandidater = [
            {"KandidatId": str(c["id"]), "Stemmer": v}
            for c, v in zip(candidates, candidate_votes)
        ]
        inden_for_parti.append({
            "PartiId": party_id,
            "Partistemmer": total,
            "Kandidater": kandidater,
        })
    return {
        "Valgresultater": {
            "AfstemningsomraadeId": str(ao_id),
            "Optaellingstype": optaellingstype,
            "IndenforParti": inden_for_parti,
            "KandidaterUdenforParti": [],
```

```python
def download_sftp_dir(remote_dir: str, local_dir: Path, force: bool = False) -> None:
    """Download all non-.hash files from remote_dir to local_dir (flat, no recurse)."""
    local_dir.mkdir(parents=True, exist_ok=True)
    ssh, sftp = _sftp_connect()
    try:
        entries = sftp.listdir(remote_dir)
        for name in entries:
            if name.endswith(".hash"):
                continue
            local_path = local_dir / name
            if local_path.exists() and not force:
                continue
            print(f"  downloading {name} …")
            sftp.get(f"{remote_dir}/{name}", str(local_path))
    finally:
        sftp.close()
        ssh.close()
```

```python
# ── Parse ─────────────────────────────────────────────────────────────────────

def parse_geografi(geografi_dir: Path) -> dict[str, dict]:
    """
    Parse FV2026 geografi files into a hierarchy dict.

    Returns {ao_id: {ao_name, ok_id, ok_name, sk_id, sk_name, eligible_voters}}
    All IDs are strings.
    """
    # Build storkreds lookup: code → name
    sk_lookup: dict[str, str] = {}
    ok_lookup: dict[str, dict] = {}  # code → {name, sk_id}

    for f in geografi_dir.glob("*.json"):
        if f.name.endswith(".hash"):
            continue
        data = json.loads(f.read_text())
        if not isinstance(data, list):
            continue
        if data and "AntalKredsmandater" in data[0]:
            # Storkreds / Region file
            for item in data:
                sk_lookup[str(item["Kode"])] = item.get("Navn", "")
        elif data and "StorkredskodeKode" in data[0]:
            # Opstillingskreds file
            for item in data:
                ok_lookup[str(item["Kode"])] = {
                    "name": item.get("Navn", ""),
                    "sk_id": str(item.get("StorkredskodeKode", "")),
                }
        # AO files handled below

    ao_hierarchy: dict[str, dict] = {}
    for f in geografi_dir.glob("*.json"):
        if f.name.endswith(".hash"):
            continue
        data = json.loads(f.read_text())
        if not isinstance(data, list) or not data:
            continue
        if "OpstillingskredsKode" in data[0]:
            # Afstemningsomraade file
            for item in data:
                ok_id = str(item.get("OpstillingskredsKode", ""))
                ok_info = ok_lookup.get(ok_id, {})
                sk_id = ok_info.get("sk_id", "")
                ao_hierarchy[str(item["Kode"])] = {
                    "ao_name": item.get("Navn", ""),
                    "ok_id": ok_id,
                    "ok_name": ok_info.get("name", ""),
                    "sk_id": sk_id,
                    "sk_name": sk_lookup.get(sk_id, ""),
                    "eligible_voters": item.get("AntalStemmeberettigede", 0) or 0,
                }

    return ao_hierarchy


def parse_kandidatdata(kandidat_dir: Path) -> dict[str, dict[str, list[dict]]]:
    """
    Parse FV2026 kandidat-data files.

    Returns {ok_id: {party_id: [{id, ballot_position}]}} sorted by ballot_position.
    ok_id is the OpstillingskredsKode from each candidate record.
    """
    result: dict[str, dict[str, list[dict]]] = {}

    for f in kandidat_dir.glob("*.json"):
        if f.name.endswith(".hash"):
            continue
        data = json.loads(f.read_text())
        valg = data.get("Valg", {})
        for party in valg.get("IndenforParti", []):
            party_id = party.get("Id")
            for k in party.get("Kandidater", []):
                ok_id = str(k.get("OpstillingskredsKode", ""))
                if not ok_id:
                    continue
                result.setdefault(ok_id, {}).setdefault(party_id, []).append({
                    "id": str(k.get("Id", "")),
                    "name": k.get("Navn", ""),
                    "ballot_position": k.get("Stemmeseddelplacering", 99),
                })

    # Sort each party's candidates by ballot position
    for ok_id in result:
        for party_id in result[ok_id]:
            result[ok_id][party_id].sort(key=lambda c: c["ballot_position"])

    return result


def parse_fv2022_csv(csv_path: Path) -> dict[tuple[str, str], dict[str, int]]:
    """
    Parse FV2022 results CSV.

    Returns {(ok_name_norm, ao_name_norm): {party_id: party_votes}}
    Only 'Partiliste' rows are counted (these are the party totals).
    """
    result: dict[tuple[str, str], dict[str, int]] = {}
    with csv_path.open(encoding="utf-8-sig", newline="") as f:
        reader = csv.DictReader(f, delimiter=";")
        for row in reader:
            if row.get("Navn", "").strip() != "Partiliste":
                continue
            ok_norm = normalize_name(row.get("Opstillingskreds", ""))
            ao_norm = normalize_name(row.get("Afstemningsområde", ""))
            party_id = row.get("Partibogstav", "").strip()
            try:
                votes = int(row.get("Stemmetal", 0))
            except (ValueError, TypeError):
                continue
            result.setdefault((ok_norm, ao_norm), {})[party_id] = votes
    return result


def build_id_mapping(hierarchy: dict[str, dict]) -> dict[tuple[str, str], str]:
    """Build {(ok_name_norm, ao_name_norm): ao_id} from the geografi hierarchy."""
    mapping: dict[tuple[str, str], str] = {}
    for ao_id, info in hierarchy.items():
        key = (normalize_name(info["ok_name"]), normalize_name(info["ao_name"]))
        mapping[key] = ao_id
    return mapping
```

```python
# ── Wave assignment ────────────────────────────────────────────────────────────

N_PRELIM_WAVES = 25   # wave_01 … wave_25
N_FINAL_WAVES = 7     # wave_26 … wave_32
FINTAELLING_OFFSET = 13  # fintælling waves start this many waves after their AO's prelim wave


def assign_preliminary_waves(
    matched_aos: dict[str, dict],
    n_waves: int,
    island_ao_ids: set[str],
) -> dict[str, int]:
    """
    Assign each AO an integer wave number (1-indexed).

    Island AOs → wave 1 regardless of size.
    Remaining AOs sorted by eligible_voters ascending, split into n_waves-1 equal buckets.
    Returns {ao_id: wave_number (1-based)}.
    """
    assignment: dict[str, int] = {}

    non_island = {k: v for k, v in matched_aos.items() if k not in island_ao_ids}
    island = {k: v for k, v in matched_aos.items() if k in island_ao_ids}

    for ao_id in island:
        assignment[ao_id] = 1

    sorted_aos = sorted(non_island, key=lambda k: non_island[k].get("eligible_voters", 0))
    bucket_size = max(1, math.ceil(len(sorted_aos) / (n_waves - 1)))

    for i, ao_id in enumerate(sorted_aos):
        wave = min(2 + i // bucket_size, n_waves)
        assignment[ao_id] = wave

    return assignment


def detect_island_ao_ids(hierarchy: dict[str, dict]) -> set[str]:
    """Return AO IDs that belong to island opstillingskredse by name."""
    island_ids: set[str] = set()
    for ao_id, info in hierarchy.items():
        ok_norm = normalize_name(info.get("ok_name", ""))
        sk_id = info.get("sk_id", "")
        if ok_norm in ISLAND_OK_NAMES or sk_id in ISLAND_STORKREDS_CODES:
            island_ids.add(ao_id)
    return island_ids

def write_preliminary_wave(
    wave_dir: Path,
    wave_index: int,
    ao_ids_in_wave: list[str],
    hierarchy: dict[str, dict],
    cumulative_ok_votes: dict[str, dict[str, int]],
    fv2022_votes: dict[tuple[str, str], dict[str, int]],
    id_mapping: dict[tuple[str, str], str],
    description: str,
) -> None:
    """Write one preliminary wave: partistemmefordeling (cumulative) + valgdeltagelse."""
    wave_dir.mkdir(parents=True, exist_ok=True)

    t = WAVE_TIMES[wave_index]
    (wave_dir / "_meta.json").write_text(json.dumps({
        "label": f"{t} — {description}",
        "time": t,
        "interval_s": float(WAVE_INTERVALS[wave_index]),
        "phase": "preliminary",
    }, ensure_ascii=False, indent=2))

    pf_dir = wave_dir / "partistemmefordeling"
    pf_dir.mkdir(exist_ok=True)
    vd_dir = wave_dir / "valgdeltagelse"
    vd_dir.mkdir(exist_ok=True)

    affected_ok_ids: set[str] = set()

    for ao_id in ao_ids_in_wave:
        info = hierarchy[ao_id]
        ok_id = info["ok_id"]
        ok_name = info["ok_name"]
        ao_name = info["ao_name"]
        key = (normalize_name(ok_name), normalize_name(ao_name))
        ao_party_votes = fv2022_votes.get(key, {})

        # Accumulate into cumulative ok votes
        ok_votes = cumulative_ok_votes.setdefault(ok_id, {})
        for party_id, votes in ao_party_votes.items():
            ok_votes[party_id] = ok_votes.get(party_id, 0) + votes
        affected_ok_ids.add(ok_id)

        # valgdeltagelse — minimal turnout snapshot
        eligible = info.get("eligible_voters", 0)
        total_party_votes = sum(ao_party_votes.values())
        vd_data = {
            "Valgart": "Folketingsvalg",
            "AfstemningsområdeDagiId": str(ao_id),
            "Valgdeltagelse": [{
                "AntalStemmeberettigede": eligible,   # matches valgdeltagelse plugin key
                "AfgivneStemmer": total_party_votes,
                "Tidspunkt": f"{t}:00",
            }],
        }
        (vd_dir / f"valgdeltagelse-{ao_id}.json").write_text(
            json.dumps(vd_data, ensure_ascii=False, indent=2)
        )

    # Write cumulative partistemmefordeling for all affected opstillingskredse
    for ok_id in affected_ok_ids:
        pf = build_partistemmefordeling(ok_id, cumulative_ok_votes[ok_id])
        (pf_dir / f"partistemmefordeling-{ok_id}.json").write_text(
            json.dumps(pf, ensure_ascii=False, indent=2)
        )


def write_fintaelling_wave(
    wave_dir: Path,
    wave_index: int,
    ao_ids_in_wave: list[str],
    hierarchy: dict[str, dict],
    fv2022_votes: dict[tuple[str, str], dict[str, int]],
    kandidatdata: dict[str, dict[str, list[dict]]],
    description: str,
) -> None:
    """Write one fintælling wave: valgresultater with synthetic candidate votes."""
    wave_dir.mkdir(parents=True, exist_ok=True)

    t = WAVE_TIMES[wave_index]
    (wave_dir / "_meta.json").write_text(json.dumps({
        "label": f"{t} — {description}",
        "time": t,
        "interval_s": float(WAVE_INTERVALS[wave_index]),
        "phase": "final",
    }, ensure_ascii=False, indent=2))

    vr_dir = wave_dir / "valgresultater"
    vr_dir.mkdir(exist_ok=True)

    for ao_id in ao_ids_in_wave:
        info = hierarchy[ao_id]
        ok_id = info["ok_id"]
        ok_name = info["ok_name"]
        ao_name = info["ao_name"]
        key = (normalize_name(ok_name), normalize_name(ao_name))
        ao_party_votes = fv2022_votes.get(key, {})

        party_data: dict[str, dict] = {}
        for party_id, total in ao_party_votes.items():
            candidates = kandidatdata.get(ok_id, {}).get(party_id, [])
            party_data[party_id] = {
                "total": total,
                "candidates_by_ok": {ok_id: candidates},
            }

        vr = build_valgresultater(
            ao_id=str(ao_id),
            optaellingstype="Fintaelling",
            party_data=party_data,
            ao_ok_id=ok_id,
        )
        safe_ao_name = ao_name.replace("/", "-").replace(" ", "_")[:40]
        filename = f"valgresultater-Folketingsvalg-{safe_ao_name}-{ao_id}.json"
        (vr_dir / filename).write_text(json.dumps(vr, ensure_ascii=False, indent=2))
```
